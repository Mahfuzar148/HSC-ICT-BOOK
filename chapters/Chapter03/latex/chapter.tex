\chapter{সংখ্যা পদ্ধতি ও ডিজিটাল ডিভাইস}

\begin{learningbox}
এই অধ্যায় শেষে শিক্ষার্থী পারবে:
\begin{itemize}
\item সংখ্যা পদ্ধতির ভিত্তি, digit, place value এবং positional system ব্যাখ্যা করতে।
\item binary, octal, decimal ও hexadecimal সংখ্যার মধ্যে রূপান্তর করতে।
\item signed number, 1's complement, 2's complement এবং binary arithmetic প্রয়োগ করতে।
\item BCD, ASCII, EBCDIC ও Unicode-এর কাজ ও পার্থক্য ব্যাখ্যা করতে।
\item Boolean algebra, truth table, logic gate ও মৌলিক digital circuit বিশ্লেষণ করতে।
\end{itemize}
\end{learningbox}

\begin{center}
\fbox{\parbox{0.92\textwidth}{
\centering
\textbf{অধ্যায় মানচিত্র}\\[4pt]
সংখ্যা পদ্ধতি $\rightarrow$ রূপান্তর ও binary arithmetic $\rightarrow$ code $\rightarrow$ Boolean algebra $\rightarrow$ logic gates $\rightarrow$ encoder/decoder/adder/register/counter
}}
\end{center}

\begin{keypointbox}{High-yield অংশ}
\begin{itemize}
\item Binary, octal, decimal, hexadecimal conversion এবং 2's complement।
\item BCD, ASCII, EBCDIC ও Unicode-এর তুলনা।
\item Boolean theorem, De Morgan, truth table ও logic function simplification।
\item AND, OR, NOT, NAND, NOR, XOR, XNOR gate-এর truth table।
\item Half adder, full adder, encoder, decoder, register ও counter।
\end{itemize}
\end{keypointbox}
