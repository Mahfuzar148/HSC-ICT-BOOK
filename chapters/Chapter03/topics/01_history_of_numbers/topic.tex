% ৩.১ সংখ্যা আবিষ্কারের ইতিহাস
\section{সংখ্যা আবিষ্কারের ইতিহাস}

\subsection{সংখ্যার প্রয়োজনীয়তা}
\begin{itemize}
\item গণনা ও ব্যবসা, মাপজোক, সময়-পরিমাপ
\end{itemize}

\subsection{গণনার প্রাচীন পদ্ধতি}
\begin{itemize}
\item আঙুল, পাথর, দাগ, কাঠি ব্যবহার
\item বিভিন্ন সভ্যতায় সংখ্যা ব্যবহারের উদাহরণ
\end{itemize}

\subsection{আধুনিক সংখ্যা পদ্ধতির বিকাশ}
\begin{itemize}
\item Hindu–Arabic সংখ্যা পদ্ধতির উন্নয়ন
\item ডিজিটাল প্রযুক্তিতে সংখ্যার গুরুত্ব
\end{itemize}

\begin{learningbox}
এই টপিক শেষে শিক্ষার্থী পারবে:
\begin{itemize}
\item সংখ্যা ব্যবহারের প্রয়োজনীয়তা ব্যাখ্যা করতে।
\item প্রাচীন গণনা পদ্ধতি থেকে আধুনিক positional number system-এর ধারণা বুঝতে।
\item ডিজিটাল ডিভাইসে সংখ্যার ভূমিকা লিখতে।
\end{itemize}
\end{learningbox}

\begin{definitionbox}{সংখ্যা}
যে প্রতীক বা প্রতীকের সমষ্টি দিয়ে গণনা, পরিমাপ বা পরিমাণ প্রকাশ করা হয়, তাকে সংখ্যা বলা হয়। সংখ্যা মানুষের হিসাব-নিকাশকে নির্দিষ্ট ও লিখিত রূপ দেয়।
\end{definitionbox}

\begin{keypointbox}{মূল ধারণা}
\begin{itemize}
\item মানুষ প্রথমে দৈনন্দিন প্রয়োজন যেমন পশু গণনা, শস্য মাপা, সময় ও বাণিজ্যের জন্য সংখ্যা ব্যবহার করত।
\item positional system আসার পর একই digit অবস্থানভেদে ভিন্ন মান প্রকাশ করতে পারে।
\item computer সব তথ্যকে শেষ পর্যন্ত binary digit বা bit হিসেবে প্রক্রিয়াকরণ করে।
\end{itemize}
\end{keypointbox}

\begin{examplebox}{বাস্তব উদাহরণ}
দোকানের বিক্রয় হিসাব decimal সংখ্যায় লেখা হলেও computer billing system সেই সংখ্যাকে memory-তে binary form-এ সংরক্ষণ ও প্রক্রিয়াকরণ করে।
\end{examplebox}

\item positional number system কেন গুরুত্বপূর্ণ?
