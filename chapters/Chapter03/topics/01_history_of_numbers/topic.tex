\section{সংখ্যা আবিষ্কারের ইতিহাস}

\subsection{সংখ্যার প্রয়োজনীয়তা}
\begin{itemize}
\item গণনা ও ব্যবসা, মাপজোক, সময়-পরিমাপ
\end{itemize}

\subsection{গণনার প্রাচীন পদ্ধতি}
\begin{itemize}
\item আঙুল, পাথর, দাগ, কাঠি ব্যবহার
\item বিভিন্ন সভ্যতায় সংখ্যা ব্যবহারের উদাহরণ
\end{itemize}

\subsection{আধুনিক সংখ্যা পদ্ধতির বিকাশ}
\begin{itemize}
\item Hindu–Arabic সংখ্যা পদ্ধতির উন্নয়ন
\item ডিজিটাল প্রযুক্তিতে সংখ্যার গুরুত্ব
\end{itemize}
