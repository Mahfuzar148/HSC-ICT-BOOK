\section{BCD, EBCDIC, ASCII এবং Unicode}

\subsection{BCD}
\begin{itemize}
\item Binary Coded Decimal: প্রতিটি decimal digit-কে 4-bit দিয়ে প্রকাশ
\end{itemize}

\subsection{EBCDIC}
\begin{itemize}
\item IBM-এর 8-bit code
\end{itemize}

\subsection{ASCII}
\begin{itemize}
\item 7-bit বা extended 8-bit; English character representation
\end{itemize}

\subsection{Unicode}
\begin{itemize}
\item বহুভাষিক character representation; বাংলা ও অন্যান্য ভাষা
\end{itemize}
