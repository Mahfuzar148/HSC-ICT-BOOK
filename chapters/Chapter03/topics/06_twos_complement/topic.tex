\section{2's Complement}

\subsection{Complement ধারণা}
\begin{itemize}
\item 1's complement কী
\item 2's complement কী
\end{itemize}

\subsection{2's complement বের করার নিয়ম}
\begin{enumerate}
\item প্রতিটি bit উল্টাও (1's complement)
\item সেখানে 1 যোগ করো
\end{enumerate}

\subsection{ব্যবহার}
\begin{itemize}
\item ঋণাত্মক সংখ্যা প্রকাশ
\item Binary subtraction সহজ করা
\end{itemize}
