\chapter{কমিউনিকেশন সিস্টেমস ও নেটওয়ার্কিং}

\begin{learningbox}
এই অধ্যায় শেষে শিক্ষার্থী পারবে:
\begin{itemize}
\item ডেটা কমিউনিকেশন সিস্টেমের উপাদান, ব্যান্ডউইডথ এবং সিগন্যাল ব্যাখ্যা করতে।
\item ডেটা ট্রান্সমিশন মেথড, মোড এবং কমিউনিকেশন মিডিয়ার শ্রেণিবিভাগ করতে।
\item তারযুক্ত ও তারবিহীন মাধ্যমের ব্যবহার, সুবিধা ও সীমাবদ্ধতা তুলনা করতে।
\item মোবাইল প্রজন্ম, কম্পিউটার নেটওয়ার্ক, নেটওয়ার্ক ডিভাইস ও টপোলজি বিশ্লেষণ করতে।
\item ক্লাউড কম্পিউটিংয়ের ধারণা, সেবা মডেল এবং বাস্তব ব্যবহার ব্যাখ্যা করতে।
\end{itemize}
\end{learningbox}

\begin{center}
\fbox{\parbox{0.92\textwidth}{
\centering
\textbf{অধ্যায় মানচিত্র}\\[4pt]
ডেটা কমিউনিকেশন $\rightarrow$ ট্রান্সমিশন মেথড/মোড $\rightarrow$ মিডিয়া $\rightarrow$ মোবাইল ও নেটওয়ার্ক $\rightarrow$ ডিভাইস/টপোলজি $\rightarrow$ ক্লাউড কম্পিউটিং
}}
\end{center}

\begin{keypointbox}{High-yield অংশ}
\begin{itemize}
\item ব্যান্ডউইডথের শ্রেণি: Narrow band, Voice band, Broadband।
\item ট্রান্সমিশন মেথড: Parallel, Serial; Serial-এর মধ্যে Asynchronous, Synchronous, Isochronous।
\item ট্রান্সমিশন মোড: Simplex, Half duplex, Full duplex; delivery mode: Unicast, Broadcast, Multicast।
\item মিডিয়া: Guided/Wired এবং Unguided/Wireless।
\item নেটওয়ার্ক: PAN, LAN, CAN, MAN, WAN; topology: Bus, Ring, Star, Tree, Mesh, Hybrid।
\end{itemize}
\end{keypointbox}
