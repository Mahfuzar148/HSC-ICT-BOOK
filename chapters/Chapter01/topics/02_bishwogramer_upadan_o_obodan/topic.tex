\begin{learningbox}

\begin{itemize}
\item বিশ্বগ্রাম প্রতিষ্ঠার প্রধান উপাদানগুলো চিহ্নিত করতে।

\item শিক্ষা, চিকিৎসা, ব্যবসা, গবেষণা ও কর্মসংস্থানে ICT-এর অবদান লিখতে।
\end{itemize}
\end{learningbox}
