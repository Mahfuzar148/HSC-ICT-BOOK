\section{Practical Program Practice}

\begin{learningbox}
এই টপিক শেষে শিক্ষার্থী HSC board-style basic C program লিখে output ব্যাখ্যা করতে পারবে।
\end{learningbox}

\begin{practicebox}
\textbf{Program 1: দুই সংখ্যার যোগফল}
\begin{verbatim}
#include <stdio.h>
int main()
{
    int a, b, sum;
    scanf("%d %d", &a, &b);
    sum = a + b;
    printf("%d", sum);
    return 0;
}
\end{verbatim}
\end{practicebox}

\begin{practicebox}
\textbf{Program 2: জোড়-বিজোড়}
\begin{verbatim}
if (n % 2 == 0)
    printf("Even");
else
    printf("Odd");
\end{verbatim}
\end{practicebox}

\begin{keypointbox}{প্র্যাকটিস তালিকা}
\begin{itemize}
\item তিন সংখ্যার মধ্যে বড় সংখ্যা।
\item 1 থেকে n পর্যন্ত যোগফল।
\item factorial।
\item multiplication table।
\item array-এর sum ও average।
\item function ব্যবহার করে যোগফল।
\end{itemize}
\end{keypointbox}
