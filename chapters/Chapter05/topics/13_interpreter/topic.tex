\section{Interpreter}

\subsection{Interpreter কী}
\begin{itemize}
\item Line by line translate করে, error পেলে execution থামে
\item object program না-ও তৈরি হতে পারে
\end{itemize}

\subsection{সুবিধা ও অসুবিধা}
\begin{itemize}
\item সুবিধা: error correction সহজ, শিক্ষার জন্য উপযোগী
\item অসুবিধা: execution ধীর, প্রতিবার run এ translate প্রয়োজন
\end{itemize}
