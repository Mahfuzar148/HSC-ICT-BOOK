% ১১. অ্যালগরিদম
\section{অ্যালগরিদম}

\begin{learningbox}
এই টপিক শেষে শিক্ষার্থী algorithm-এর বৈশিষ্ট্য ও উদাহরণ লিখতে পারবে।
\end{learningbox}

\begin{definitionbox}{Algorithm}
কোনো সমস্যা সমাধানের জন্য সীমিত সংখ্যক ধাপে সাজানো সুস্পষ্ট ও যৌক্তিক নির্দেশসমষ্টিকে algorithm বলা হয়।
\end{definitionbox}

\begin{keypointbox}{ভালো algorithm-এর বৈশিষ্ট্য}
\begin{itemize}
\item শুরু ও শেষ থাকবে।
\item প্রতিটি ধাপ স্পষ্ট হবে।
\item ধাপ সংখ্যা সীমিত হবে।
\item input ও output নির্দিষ্ট থাকবে।
\item logic সঠিক ও কার্যকর হবে।
\end{itemize}
\end{keypointbox}

\begin{examplebox}{দুই সংখ্যার যোগফলের algorithm}
Step 1: Start। Step 2: দুটি সংখ্যা A, B input নাও। Step 3: Sum = A + B। Step 4: Sum print কর। Step 5: Stop।
\end{examplebox}

\begin{practicebox}
\begin{enumerate}
\item Algorithm কী?
\item একটি সংখ্যা জোড় না বিজোড় নির্ণয়ের algorithm লেখ।
\end{enumerate}
\end{practicebox}
