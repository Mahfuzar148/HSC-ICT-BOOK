% ১২. ফ্লোচার্ট
\section{ফ্লোচার্ট}

\begin{learningbox}
এই টপিক শেষে শিক্ষার্থী flowchart-এর symbol, ব্যবহার এবং algorithm-এর সাথে সম্পর্ক ব্যাখ্যা করতে পারবে।
\end{learningbox}

\begin{definitionbox}{Flowchart}
Algorithm বা problem-solving steps-কে নির্দিষ্ট symbol ও arrow ব্যবহার করে চিত্রের মাধ্যমে প্রকাশ করলে তাকে flowchart বলা হয়।
\end{definitionbox}

\begin{center}
\begin{tabular}{|p{0.22\textwidth}|p{0.34\textwidth}|p{0.28\textwidth}|}
\hline
\textbf{Symbol} & \textbf{কাজ} & \textbf{আকৃতি} \\
\hline
Terminal & Start/End & Oval \\
\hline
Input/Output & data নেওয়া/দেখানো & Parallelogram \\
\hline
Process & calculation/assignment & Rectangle \\
\hline
Decision & condition test & Diamond \\
\hline
Flow line & direction দেখায় & Arrow \\
\hline
\end{tabular}
\end{center}

\begin{center}
\fbox{\parbox{0.88\textwidth}{
\centering
\textbf{Flowchart placeholder}\\[4pt]
Start $\rightarrow$ Input A,B $\rightarrow$ Sum=A+B $\rightarrow$ Print Sum $\rightarrow$ Stop
}}
\end{center}

\begin{mistakebox}{সাধারণ ভুল}
Decision symbol ছাড়া if/condition দেখালে flowchart অসম্পূর্ণ হয়। Yes/No branch label দিতে হবে।
\end{mistakebox}
