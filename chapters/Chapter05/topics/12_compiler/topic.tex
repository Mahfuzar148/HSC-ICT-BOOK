\section{Compiler}

\subsection{Compiler কী}
\begin{itemize}
\item সম্পূর্ণ program একসাথে translate করে
\item source → object এবং error list একসঙ্গে দেখায়
\end{itemize}

\subsection{সুবিধা ও অসুবিধা}
\begin{itemize}
\item সুবিধা: execution দ্রুত, object code তৈরি
\item অসুবিধা: debugging ধীর, memory বেশি লাগতে পারে
\end{itemize}
