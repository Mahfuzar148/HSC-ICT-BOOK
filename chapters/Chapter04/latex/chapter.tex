\chapter{ওয়েব ডিজাইন পরিচিতি এবং HTML}

\begin{learningbox}
এই অধ্যায় শেষে শিক্ষার্থী পারবে:
\begin{itemize}
\item ওয়েব পেজ, ওয়েবসাইট, ব্রাউজার, সার্ভার ও ওয়েব ডিজাইনের ধারণা ব্যাখ্যা করতে।
\item website structure, navigation, layout এবং responsive design বুঝতে।
\item HTML document structure, tag, attribute, element ও syntax ব্যবহার করতে।
\item heading, paragraph, formatting, hyperlink, image, table ও form তৈরি করতে।
\item একটি basic website তৈরি, test এবং publish করার ধাপ লিখতে।
\end{itemize}
\end{learningbox}

\begin{center}
\fbox{\parbox{0.92\textwidth}{
\centering
\textbf{অধ্যায় মানচিত্র}\\[4pt]
Web concept $\rightarrow$ Webpage/Website $\rightarrow$ Web terms $\rightarrow$ Website types/structure $\rightarrow$ HTML basics $\rightarrow$ Text/link/image/table/form $\rightarrow$ Page design $\rightarrow$ Publishing
}}
\end{center}

\begin{keypointbox}{High-yield অংশ}
\begin{itemize}
\item Static vs dynamic website, webpage vs website, client-server ধারণা।
\item IP address, domain name, DNS, URL, HTTP/HTTPS, browser ও search engine।
\item HTML tag, element, attribute, paired/empty tag।
\item \texttt{<a>}, \texttt{<img>}, \texttt{<table>}, \texttt{<form>} tag-এর syntax ও attribute।
\item Linear, tree, network/hybrid website structure।
\item Domain, hosting, URL, FTP/SFTP ও website publishing।
\end{itemize}
\end{keypointbox}
