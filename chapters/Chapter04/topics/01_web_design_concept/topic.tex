\section{ওয়েব ডিজাইনের ধারণা}

\subsection{ওয়েব ডিজাইন কী}
\begin{itemize}
\item ওয়েব পেজ ও ওয়েবসাইটের সাধারণ ধারণা
\item ওয়েব ব্রাউজার ও ওয়েব সার্ভার
\end{itemize}

\subsection{ওয়েবসাইট তৈরির ধাপ ও বৈশিষ্ট্য}
\begin{itemize}
\item পরিকল্পনা, কন্টেন্ট, লেআউট, কোডিং, টেস্টিং, পাবলিশিং
\item ভালো ওয়েব পেজের বৈশিষ্ট্য
\end{itemize}
