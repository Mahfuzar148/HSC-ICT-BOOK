% ১৮. অনুশীলনী
\section{অনুশীলনী}
\begin{practicebox}
\textbf{সংক্ষিপ্ত প্রশ্ন}
\begin{enumerate}
    \item ডেটা ও তথ্যের মধ্যে পার্থক্য লেখো।
    \item ফিল্ড, রেকর্ড ও টেবিল কী?
    \item DBMS-এর চারটি কাজ লেখো।
    \item RDBMS বলতে কী বোঝায়?
    \item Primary key ও Foreign key-এর পার্থক্য লেখো।
    \item Query কেন ব্যবহার করা হয়?
    \item Sorting ও Indexing-এর পার্থক্য লেখো।
    \item Form ও Report-এর ব্যবহার লেখো।
    \item কর্পোরেট ডেটাবেজের দুটি উদাহরণ দাও।
    \item ডেটা এনক্রিপশন কেন প্রয়োজন?
\end{enumerate}
\end{practicebox}

\begin{boardbox}{সৃজনশীল প্রশ্নের প্রস্তুতি}
একটি কলেজের Student, Result ও Payment টেবিল দেওয়া থাকলে কোন ফিল্ড Primary key হবে, কোন ফিল্ড Foreign key হবে, কীভাবে One-to-Many relation তৈরি হবে এবং নির্দিষ্ট শর্তের Query কীভাবে লেখা হবে--এসব ব্যাখ্যা করার অনুশীলন করো।
\end{boardbox}

\begin{recapbox}
Chapter 6-এর মূল কথা: ডেটাকে সংগঠিতভাবে রাখার জন্য ডেটাবেজ, ডেটাবেজ পরিচালনার জন্য DBMS, সম্পর্কভিত্তিক টেবিল ব্যবস্থার জন্য RDBMS, ডেটা বের করার জন্য Query, দ্রুত অনুসন্ধানের জন্য Indexing, উপস্থাপনের জন্য Form/Report এবং নিরাপত্তার জন্য Access control, Backup ও Encryption ব্যবহৃত হয়।
\end{recapbox}
