\section{ডেটাবেজ ম্যানেজমেন্ট সিস্টেম --- পরিচিতি}
বাস্তব জীবনের যেকোনো প্রতিষ্ঠানে ছাত্র, শিক্ষক, কর্মচারী, পণ্য, বিক্রয়, ব্যাংক হিসাব, রোগীর তথ্য বা নাগরিক সেবার তথ্য নিয়মিত জমা হয়। এই তথ্যগুলো যদি এলোমেলোভাবে রাখা হয়, তাহলে দ্রুত খুঁজে পাওয়া, সংশোধন করা বা প্রতিবেদন তৈরি করা কঠিন হয়। ডেটাবেজ এই সমস্যার সংগঠিত সমাধান।

\begin{definitionbox}{ডেটা, তথ্য ও ডেটাবেজ}
\textbf{ডেটা} হলো কাঁচা উপাত্ত, যেমন নাম, রোল, নম্বর, তারিখ। ডেটা প্রক্রিয়াকরণের পর অর্থবহ ফল পাওয়া গেলে সেটি \textbf{তথ্য}। সম্পর্কযুক্ত ডেটাকে নিয়ম মেনে সাজিয়ে সংরক্ষণ করার সংগঠিত সংগ্রহকে \textbf{ডেটাবেজ} বলে।
\end{definitionbox}

\begin{keypointbox}{ডেটা থেকে সিদ্ধান্ত}
\begin{center}
\fbox{\parbox{0.82\textwidth}{\centering
ডেটা সংগ্রহ $\rightarrow$ ডেটা সংরক্ষণ $\rightarrow$ অনুসন্ধান/প্রক্রিয়াকরণ $\rightarrow$ তথ্য $\rightarrow$ সিদ্ধান্ত}}
\end{center}
\textbf{চিত্রের স্থানধারক:} ডেটা থেকে তথ্য তৈরির প্রবাহচিত্র।
\end{keypointbox}

\begin{examplebox}{শিক্ষাপ্রতিষ্ঠানের উদাহরণ}
একটি কলেজে ছাত্রের রোল, নাম, বিভাগ, মোবাইল নম্বর, ভর্তি সাল এবং ফলাফল আলাদা আলাদা খাতায় না রেখে একটি ডেটাবেজে রাখা হলে দ্রুত ভর্তি তালিকা, পরীক্ষার ফল, বকেয়া ফি বা উপস্থিতি রিপোর্ট বের করা যায়।
\end{examplebox}

\begin{keypointbox}{ডেটাবেজ ব্যবহারের ক্ষেত্র}
ব্যাংকিং, হাসপাতাল, শিক্ষা প্রতিষ্ঠান, ই-কমার্স, রেল/এয়ার টিকেটিং, মোবাইল অপারেটর, নির্বাচন কমিশন, পাসপোর্ট অফিস, ভূমি রেকর্ড এবং কর ব্যবস্থায় ডেটাবেজ অপরিহার্য।
\end{keypointbox}

\begin{boardbox}{বোর্ড ফোকাস}
ডেটাবেজের মূল উদ্দেশ্য হলো ডেটা সংরক্ষণ, দ্রুত অনুসন্ধান, পুনরাবৃত্তি কমানো, তথ্যের সামঞ্জস্য রক্ষা এবং প্রয়োজন অনুযায়ী রিপোর্ট তৈরি করা।
\end{boardbox}
