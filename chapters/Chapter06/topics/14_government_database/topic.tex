% ১৪. সরকারি প্রতিষ্ঠানে ডেটাবেজ
\section{সরকারি প্রতিষ্ঠানে ডেটাবেজ}
সরকারি ডেটাবেজ নাগরিক সেবা, প্রশাসনিক কাজ, নীতি নির্ধারণ এবং স্বচ্ছতা বৃদ্ধিতে গুরুত্বপূর্ণ ভূমিকা রাখে। ডিজিটাল বাংলাদেশ/স্মার্ট বাংলাদেশ ধারণায় সরকারি ডেটাবেজের ব্যবহার আরও বেড়েছে।

\subsection{ধারণা ও ব্যবহার}
\begin{keypointbox}{উদাহরণ}
\begin{itemize}
    \item জাতীয় পরিচয়পত্র ও ভোটার ডেটাবেজ।
    \item জন্ম-মৃত্যু নিবন্ধন ডেটাবেজ।
    \item পাসপোর্ট, ভিসা ও ইমিগ্রেশন ডেটাবেজ।
    \item ভূমি রেকর্ড ও নামজারি ডেটাবেজ।
    \item আয়কর, ভ্যাট ও রাজস্ব ডেটাবেজ।
    \item শিক্ষা বোর্ড, ফলাফল ও ভর্তি ডেটাবেজ।
    \item স্বাস্থ্য, হাসপাতাল ও টিকা ডেটাবেজ।
\end{itemize}
\end{keypointbox}

\subsection{সুবিধা}
\begin{examplebox}{ব্যবহার}
জাতীয় পরিচয়পত্র নম্বর ব্যবহার করে নাগরিকের পরিচয় যাচাই, ব্যাংক হিসাব খোলা, সিম নিবন্ধন, সরকারি ভাতা প্রদান ও অনলাইন সেবা দেওয়া সহজ হয়।
\end{examplebox}

\begin{mistakebox}{গোপনীয়তা}
সরকারি ডেটাবেজে ব্যক্তিগত তথ্য থাকে। তাই অনুমতি ছাড়া তথ্য প্রকাশ, দুর্বল পাসওয়ার্ড, অনিরাপদ শেয়ারিং এবং ব্যাকআপহীনতা বড় ঝুঁকি।
\end{mistakebox}
